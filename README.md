# JAV-Image-parser (SIRO, GANA200)

usage: jav_image_parser.py [-h] [-g] [-s] start end

positional arguments:
  start       start number
  end         end number

optional arguments:
  -h, --help  show this help message and exit
  -g, --gana  download image from gana200
  -s, --siro  download image from siro
